常用语法

#安装软件
pacman -S 包名 ...

#给所有提示设置选项
yes | sudo pacman -S 包名 ...

#只下载包，不安装
pacman -Sw 包名 ...

#清理未安装的包文件，包文件位于 /var/cache/pacman/pkg/ 目录
pacman -Sc

#清理所有的缓存文件
pacman -Scc

#删除单个软件包，保留其全部已经安装的依赖关系
pacman -R 包名

#删除指定软件包，及其所有没有被其他已安装软件包使用的依赖关系
pacman -Rs 包名

#在删除包的同时，删除所有依赖这个软件包的程序
pacman -Rsc 包名

#在删除包时不检查依赖
pacman -Rd 包名

#更新所有包 y更新数据库，yy强制更新，u升级软件
pacman -Syu

#在数据库中查询软件包，查询位置包含了包的名字和描述
pacman -Ss 关键字

#获取已安装软件包所包含文件的列表
pacman -Ql

#从本地安装软件包
pacman -U xxx.tar.gz

#显示软件包依赖树
pactree 包名

# 卸载所有无用包
pacman -c
