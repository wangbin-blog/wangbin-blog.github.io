常用命令操作
/usr/local/nginx/sbin/nginx -c /usr/local/nginx/conf/nginx.conf启动 nginx
/usr/local/nginx/sbin/nginx -s stop停止 nginx
/usr/local/nginx/sbin/nginx -s reload重启 nginx
netstat -tunlp查看端口占用
netstat -tunlpgrep
